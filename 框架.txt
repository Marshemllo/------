├── app/                  # Flask 应用的核心目录
│   ├── static/           # 存放静态文件 (CSS, JS, 图像)
│   │   ├── layui/        # layui 框架文件 (从layui官网下载)
│   │   ├── css/          # 自定义 CSS 文件
│   │   ├── js/           # 自定义 JS 文件
│   │   └── images/       # 图片资源
│   ├── templates/        # 存放 HTML 模板文件
│   │   ├── index.html    # 首页或登录页
│   │   └── layout.html   # 基础布局/后台主框架
│   ├── __init__.py       # 应用初始化文件 (创建Flask应用实例)
│   ├── models.py         # 数据库模型定义 (SQLite 对应的数据结构)
│   ├── views.py          # 路由和业务逻辑处理 (连接URL和函数)
│   └── database.py       # 数据库连接和操作的工具函数 (可选，用于分离数据库逻辑)
├── venv/                 # Python 虚拟环境 (推荐)
├── config.py             # 应用配置 (密钥、数据库路径等)
├── run.py                # 应用启动文件
└── requirements.txt      # 依赖库列表 (Flask, Gunicorn/Waitress等)
